//	--------------------------------------------------
//	Numerical Controlled Oscillator
//	Hz of o_gen_clk = Clock Hz / num
//	--------------------------------------------------
module	nco(	      
		o_gen_clk,
		i_nco_num,
		clk,
		rst_n);

output		o_gen_clk	;	// 1Hz CLK

input	[31:0]	i_nco_num	;
input		clk		;	// 50Mhz CLK
input		rst_n		;

reg	[31:0]	cnt		;
reg		o_gen_clk	;

always @(posedge clk or negedge rst_n) begin
	if(rst_n == 1'b0) begin
		cnt		<= 32'd0;
		o_gen_clk	<= 1'd0	;
	end else begin
		if(cnt >= i_nco_num/2-1) begin
			cnt 	<= 32'd0;
			o_gen_clk	<= ~o_gen_clk;
		end else begin
			cnt <= cnt + 1'b1;
		end
	end
end

endmodule

module	nco_worksong(	      
			o_gen_clk,
			i_nco_num_worksong,
			clk,
			rst_n);

output			o_gen_clk		;	// 1Hz CLK

input	[31:0]		i_nco_num_worksong	;
input			clk			;	// 50Mhz CLK
input			rst_n			;

reg	[31:0]		cnt_worksong	;
reg			o_gen_clk	;

always @(posedge clk or negedge rst_n) begin
	if(rst_n == 1'b0) begin
		cnt_worksong 	<= 32'd0;
		o_gen_clk	<= 1'd0	;
	end else begin
		if(cnt_worksong>= i_nco_num_worksong/9-1) begin
			cnt_worksong 	<= 32'd0;
			o_gen_clk	<= ~o_gen_clk;
		end else begin
			cnt_worksong <= cnt_worksong + 1'b1;
		end
	end
end

endmodule

//	--------------------------------------------------
//	HMS(Hour:Min:Sec) Counter
//	--------------------------------------------------
module	hms_cnt(
		o_hms_cnt,
		o_max_hit,
		i_max_cnt,
		clk,
		rst_n);

output	[5:0]	o_hms_cnt		;
output		o_max_hit		;

input	[5:0]	i_max_cnt		;
input		clk			;
input		rst_n			;

reg	[5:0]	o_hms_cnt		;
reg		o_max_hit		;
always @(posedge clk or negedge rst_n) begin
	if(rst_n == 1'b0) begin
		o_hms_cnt <= 6'd0;
		o_max_hit <= 1'b0;
	end else begin
		if(o_hms_cnt >= i_max_cnt) begin
			o_hms_cnt <= 6'd0;
			o_max_hit <= 1'b1;
		end else begin
			o_hms_cnt <= o_hms_cnt + 1'b1;
			o_max_hit <= 1'b0;
		end
	end
end

endmodule
//	--------------------------------------------------
//	HMS(Hour:Min:Sec) Countdown _ Timer
//	--------------------------------------------------
module 	hms_cnt_timer(
			o_hms_cnt,
			o_min_hit,
			i_max_cnt,
			i_min_cnt,
			i_timer_mode,
			clk,
			rst_n);

output 	[5:0]	 o_hms_cnt	; // setup->count up, timer->count down
output	  	 o_min_hit	; // min_hit

input	[5:0]	i_max_cnt	; // setup
input	[5:0]	i_min_cnt	;
input	[1:0]	i_timer_mode	;

input		clk		;
input		rst_n		;

reg	[5:0]	o_hms_cnt	;
reg		o_min_hit		;

always @ (posedge clk or negedge rst_n) begin
	if(rst_n == 1'b0) begin
		o_hms_cnt <= 6'd0	;
		o_min_hit  <= 1'b0	; // off timer

	end else begin
		case(i_timer_mode)
		2'b00 : begin // timer initial
			o_hms_cnt <= 6'd0;
			o_min_hit <= 1'b0; // timer initial 00:00:00
		end
		2'b01 : begin // timer setup??
			if(o_hms_cnt >= i_max_cnt) begin
				o_hms_cnt <= 6'd0;
			end else begin
				o_hms_cnt <= o_hms_cnt +1'b1; // act same as clock
			end
		end
		2'b10 : begin // timer start
			
			if( o_hms_cnt == 6'd0) begin
				o_hms_cnt <= 6'd0 ;
				o_min_hit <= 1'b1 ;
				o_hms_cnt <= i_min_cnt - 1'b1;
			end else begin 
				o_hms_cnt <= o_hms_cnt -1'b1;
				o_min_hit  <= 1'b0	;
			end
		end
		endcase
	end
end
endmodule

module  debounce(
		o_sw,
		i_sw,
		clk);
output		o_sw			;

input		i_sw			;
input		clk			;

reg		dly1_sw			;
always @(posedge clk) begin
	dly1_sw <= i_sw;
end

reg		dly2_sw			;
always @(posedge clk) begin
	dly2_sw <= dly1_sw;
end

assign		o_sw = dly1_sw | ~dly2_sw;

endmodule

			o_timer_sec_clk= 1'b0		;
			o_timer_min_clk	 = 1'b0		;
			o_timer_hour_clk = 1'b0		; 
		end
	endcase
end

	case(i_mode)
		MODE_CLOCK: 	begin
			o_sec	= sec;
			o_min	= min;

			o_timer_min_clk	= i_timer_min_clk	;
			o_timer_sec_clk	= i_timer_sec_clk	;*/
		end
		MODE_STOPWATCH:	begin
			o_sec	= stw_sec;
			o_min	= stw_min;
			o_hour	= stw_hour;
			/*o_timer_hour_clk = i_timer_hour_clk	;
			o_timer_min_clk	= i_timer_min_clk	;
			o_timer_sec_clk	= i_timer_sec_clk	;*/
		end
		MODE_WORLDTIME: begin
		case(i_wt_change_position)
			POS_NYC : begin
			if(hour <= 6'd10) begin
				o_sec	= sec;
				o_min	= min;
				o_hour 	= hour + 6'd14; // +14 
				/*o_timer_hour_clk = i_timer_hour_clk	;
				o_timer_min_clk	= i_timer_min_clk	;

				o_min	= min;

				/*o_timer_hour_clk = i_timer_hour_clk	;
				o_timer_min_clk	= i_timer_min_clk	;
				o_timer_sec_clk	= i_timer_sec_clk	;*/
			end
			end
			POS_BEIJING: begin
			if(hour <= 6'd1) begin
				o_sec  	= sec;
				o_min	= min;

				o_timer_min_clk	= i_timer_min_clk	;
				o_timer_sec_clk	= i_timer_sec_clk	;*/
			end

				o_sec  	= sec;
				o_min	= min;
				o_hour	= hour - 6'd1; //-1
				/*o_timer_hour_clk = i_timer_hour_clk	;
				o_timer_min_clk	= i_timer_min_clk	;
				o_timer_sec_clk	= i_timer_sec_clk	;*/
			end
			end
			POS_LONDON: begin
			if(hour <= 6'd9) begin
				o_sec  	= sec;
				o_min	= min;
				o_hour	= hour + 6'd15; //24-9
				/*o_timer_hour_clk = i_timer_hour_clk	;
				o_timer_min_clk	= i_timer_min_clk	;
				o_timer_sec_clk	= i_timer_sec_clk	;*/
			end
			if(hour >= 6'd9) begin
				o_sec  	= sec;
				o_min	= min;
				o_hour	= hour - 6'd9; //-9
				/*o_timer_hour_clk = i_timer_hour_clk	;
				o_timer_min_clk	= i_timer_min_clk	;
				o_timer_sec_clk	= i_timer_sec_clk	;*/
			end
			end
			POS_PARIS: begin	
			if(hour <= 6'd8) begin
				o_sec  	= sec;
				o_min	= min;
				o_hour	= hour + 6'd16; //24-8
				/*o_timer_hour_clk = i_timer_hour_clk	;
				o_timer_min_clk	= i_timer_min_clk	;
				o_timer_sec_clk	= i_timer_sec_clk	;*/
			end
			if(hour >= 6'd8) begin
				o_sec  	= sec;
				o_min	= min;
				o_hour	= hour - 6'd8; //-8
				/*o_timer_hour_clk = i_timer_hour_clk	;
				o_timer_min_clk	= i_timer_min_clk	;
				o_timer_sec_clk	= i_timer_sec_clk	;*/
			end
			end
		endcase
		end
		MODE_TIMER: begin
			if( ((timer_sec == 6'd0) && (timer_min == 6'd0) && (timer_hour==6'd0)) && (i_timer_en == 2'b10) )begin
				o_sec  = 6'd0	;
				o_min  = 6'd0	;
				o_hour = 6'd0	;

				o_timer_min_clk	= ~i_timer_min_clk;
				o_timer_sec_clk	= ~i_timer_sec_clk;
				o_rst_n	= ~rst_n;*/
			end else begin
			o_sec = timer_sec;
			o_min = timer_min;
			o_hour= timer_hour;
			/*o_timer_hour_clk = i_timer_hour_clk	;
			o_timer_min_clk	= i_timer_min_clk	;
			o_timer_sec_clk	= i_timer_sec_clk	;*/
			end
		end	
	endcase
end

reg		o_alarm		;
always @ (posedge clk or negedge rst_n) begin
	if (rst_n == 1'b0) begin
		o_alarm <= 1'b0;
	end else begin
		if( (sec == alarm_sec) && (min == alarm_min) && (hour == alarm_hour)) begin
			o_alarm <= 1'b1 & i_alarm_en;
		end else begin
			o_alarm <= o_alarm & i_alarm_en;
		end
	end
end

			o_timer_buzz <= 1'b1 & i_timer_buzz_en;
		end else begin

		end
	end
